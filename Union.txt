package Solution;


import java.util.HashSet;

public class Union {
	public static HashSet<Integer> Union(int[]a,int []b) {
HashSet<Integer>Set=new HashSet();
for(int i:a ) {
	Set.add(i);
}
for(int j:b) {
	Set.add(j);
}
return Set;
	}	public static void main(String[] args) {
		int []a= {1,2,4,5};
		int[]b= {1,2,3};
		HashSet<Integer>Set=Union.Union(a,b);
		for(int val:Set) {
			System.out.println(val);
		}
	}}
