package Solution;
public class FindMissingNo {
	private static int actualSum(int []a,int N) {
		int actualSum=0;
		int sum=(N*(N+1))/ 2;
		for(int i = 0;i<N-1;i++) {
			 actualSum += a[i];
			
			
		}
		return sum-actualSum;
		

	}
	public static void main(String[] args) {
		int[]a= {1,2,4,5};
		int N=5;
	int result=FindMissingNo.actualSum(a, N) ;
	System.out.println(result);
		
	}
}
